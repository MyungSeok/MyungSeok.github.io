# jenv 로 여러 버전의 Java 설치하기

# jenv 설치

$ brew install jenv

$ jenv versions

# jenv 로 자바 설치

$ jenv install java

# jenv 초기화

~/.bash_profile 의 스크립트에서 아래 코드를 추가하여 jenv 를 초기화시킨다.

if which jenv > /dev/null; then eval "$(jenv init -)"; fi

bash_profile 적용

$ source ~/.bash_profile

# jenv 버전 선택

$ jenv local {버전 선택}

$ jenv global {버전 선택}
