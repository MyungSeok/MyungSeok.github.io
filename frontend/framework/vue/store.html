# 동기식 이벤트 전달

store.commit('syncFunc');

new Vuex.Store({
  state: {},
  mutations: {
    // 반드시 함수형이여야 한다.
    syncFunc: (state) => {
      /* statement */
    }
  }
});

# 비동기 이벤트 전달

store.dispatch('asyncFunc');

new Vuex.Store({
  state: {},
  actions: {
    // 반드시 함수형이여야 하며 클로져나 스코프체인의 보장이 어렵다.
    asyncFunc: (state) => {
      /* statement */
    }
  }
});
